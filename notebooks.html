def d2(i,j):
    q = 0
    r = 0
    s = 0
    t = 0
    for col in ["Survived","Sex"]:
        if data[col][i] == 1 and data[col][j] == 1:
            q += 1

            r += 1

            s += 1

            t += 1

    for col in [hasil1, hasil2]:

def iris(min, max):
  k = 4
  w = (max - min)/k
  return (w)
